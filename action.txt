name: CI/CD - Build & Push Image + Deploy

on:
  push:
    branches:
      - main
      - master

jobs:

  build-and-push:
    name: Build and Push Docker Image
    runs-on: ubuntu-latest

    steps:
      # Step 1: Checkout code
      - name: Checkout code
        uses: actions/checkout@v3

      # Step 2: Set up JDK (for Maven)
      - name: Set up JDK 21
        uses: actions/setup-java@v3
        with:
          java-version: '21'
          distribution: 'temurin'

      # Step 3: Build Project + Create Artifact
      - name: Build project
        run: |
          mvn clean package

      # Step 4: Upload build artifacts to GitHub Actions
      - name: Upload Artifact
        uses: actions/upload-artifact@v4
        with:
          name: build-artifact
          path: target/*.jar

      # Step 5: Login to Docker Hub
      - name: Login to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      # Step 6: Build Docker image
      - name: Build Docker image
        run: |
          docker build -t ${{ secrets.DOCKERHUB_USERNAME }}/myapp:${{ github.run_number }} .
          docker tag ${{ secrets.DOCKERHUB_USERNAME }}/myapp:${{ github.run_number }} \
                     ${{ secrets.DOCKERHUB_USERNAME }}/myapp:latest

      # Step 7: Push image to DockerHub
      - name: Push Docker Image
        run: |
          docker push ${{ secrets.DOCKERHUB_USERNAME }}/myapp:${{ github.run_number }}
          docker push ${{ secrets.DOCKERHUB_USERNAME }}/myapp:latest


  deploy:
    name: Deploy Container on Self-Hosted Runner
    runs-on: self-hosted
    needs: build-and-push

    steps:
      # Step 8: Pull Docker image from DockerHub
      - name: Pull Image from Docker Hub
        run: |
          docker pull ${{ secrets.DOCKERHUB_USERNAME }}/myapp:latest

      # Step 9: Stop existing container (if exists) and remove old container
      - name: Remove Old Container
        run: |
          docker stop myapp_container || true
          docker rm myapp_container || true

      # Step 10: Run new Docker container
      - name: Run New Container
        run: |
          docker run -d --name myapp_container -p 5081:8080 \
          ${{ secrets.DOCKERHUB_USERNAME }}/myapp:latest
